out_dir = 'dataset/SM_NQ_selected35'

data_path = os.path.join(base_dir, 'SM_NQ-30-minutes_05cm.csv')

    if c not in data.columns:

loc_ids = set(loc['station_id'].astype(str).tolist())
data_ids = set(data.columns.tolist())

    raise ValueError('Selected station list is invalid')
